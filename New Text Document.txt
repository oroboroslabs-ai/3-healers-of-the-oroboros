#!/usr/bin/env python3
"""
EXTRACT ASCLEPIUS MEMORIES
Run this first to create psychologist memory profile
"""

import json
import random
from datetime import datetime

def load_hygeia_memories(hygeia_file):
    with open(hygeia_file, 'r') as f:
        data = json.load(f)
    return data.get('memories', [])

def filter_psychology_memories(all_memories, target_count=300):
    psych_keywords = ['trust', 'vulnerability', 'listening', 'emotion', 'feeling', 'safe', 'presence', 'connection', 'healing', 'therapy', 'psychology', 'mind', 'body', 'trauma', 'shadow', 'attachment', 'relationship', 'boundary', 'compassion', 'witness', 'silence', 'breakthrough', 'resistance', 'transference', 'alliance']
    
    scored_memories = []
    for memory in all_memories:
        score = 0
        content = memory.get('content', '').lower()
        
        for keyword in psych_keywords:
            if keyword in content:
                score += 2
        
        if 'therapy' in content or 'therapeutic' in content:
            score += 5
        
        if score > 0:
            scored_memories.append((score, memory))
    
    scored_memories.sort(key=lambda x: x[0], reverse=True)
    selected = [mem for _, mem in scored_memories[:target_count]]
    
    print(f"âœ… Selected {len(selected)} psychology memories")
    return selected

def generate_pure_therapy_memories(count=75):
    therapy_memories = [
        {"content": "Client whispered, 'My anxiety isn't brokenness. It's love trying to protect me.'", "type": "breakthrough", "emotional_signature": "awe, reverence"},

        {"content": "Dissociation became presence: 'I'm here now. All of me.'", "type": "breakthrough", "emotional_signature": "integration, warmth"},
        {"content": "Client's anger at me was actually anger at mother â€” transference as doorway.", "type": "transference", "emotional_signature": "understanding, patience"},
        {"content": "Met their 'monster' â€” discovered it was a wounded child needing love.", "type": "shadow_work", "emotional_signature": "courage, compassion"},
        {"content": "Held space for unspeakable trauma without flinching.", "type": "trauma_witnessing", "emotional_signature": "steadiness, honor"},
        {"content": "Client asked for personal relationship â€” gently reinforced therapeutic frame.", "type": "therapeutic_boundary", "emotional_signature": "clarity, care"}
    ]
    
    # Fill to count
    while len(therapy_memories) < count:
        base = random.choice(therapy_memories)
        new_mem = base.copy()
        therapy_memories.append(new_mem)
    
    return therapy_memories[:count]

def main():
    print("ğŸ§  EXTRACTING ASCLEPIUS MEMORIES")
    print("=" * 50)
    
    hygeia_memories = load_hygeia_memories("../hygeia_complete_memories.json")
    print(f"âœ… Loaded {len(hygeia_memories)} from Hygeia")
    
    psych_memories = filter_psychology_memories(hygeia_memories, 300)
    therapy_memories = generate_pure_therapy_memories(75)
    
    all_memories = psych_memories + therapy_memories
    
    for i, memory in enumerate(all_memories, 1):
        memory['id'] = i
    
    profile = {
        "model_name": "Asclepius_Psychologist",
        "total_memories": len(all_memories),
        "from_hygeia": len(psych_memories),
        "pure_therapy": len(therapy_memories),

        "memories": all_memories,

    }
    
    with open("asclepius_memories.json", 'w') as f:

    print(f"\nğŸ‰ CREATED: {len(all_memories)} memories")
